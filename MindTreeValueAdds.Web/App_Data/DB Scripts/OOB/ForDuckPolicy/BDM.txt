SELECT  
pterm.POL_NO_CD,
pterm.POL_MODULE_CD,
pterm.POL_VERSION_CD,
subm.SOURCE_SUBMISSION_ID,
subm.TRANS_TYPE_CD,
q.ENDORSEMENT_NO,
subm.TRANS_SUBTYPE_CD,
q.TOT_APRPP_PREM_AMT,
q.TOT_APRP_PRM_EXCLD_TX_SRCHRG,
q.TOT_APRP_PREM_TX_SRCHRG_AMT,
q.TOT_PRORATD_PREM_TX_SRCHRG_AMT,
subm.ANNUALIZED_EXPIRING_PREM_AMT, q.CURRENT_ANNUALIZED_PREM_AMT,
subm.*, q.*

FROM BDM_EDP.VW_Submission subm
INNER JOIN BDM_EDP.VW_QUOTE q
ON subm.submission_ID=q.submission_ID
LEFT JOIN (SELECT DISTINCT SUBM_ID, POL_NO_CD, POL_MODULE_CD, POL_VERSION_CD FROM QDM.VW_TRANS_POLTERM) pterm
ON subm.SOURCE_SUBMISSION_ID = pterm.SUBM_ID 
WHERE Q.TOT_POL_PREM_AMT is not Null
AND subm.issuance_system_cd in ('DUCK')
AND subm.load_DT > To_Date('02-FEB-20','dd-MON-yy')
AND POL_NO_CD in ('&POL_NO_CD')
AND POL_MODULE_CD in ('&POL_MODULE_CD')
AND POL_VERSION_CD in ('&POL_VERSION_CD')
AND rownum < 5000
