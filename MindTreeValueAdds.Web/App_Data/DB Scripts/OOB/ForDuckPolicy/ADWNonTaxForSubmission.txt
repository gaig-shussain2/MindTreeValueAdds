SELECT
PT.BUSNACTYCD, 
PT.ENDT_NO, 
PT.SYSACTYCD, 
PT.POL_NO_CD, 
PT.POL_MODULE_CD, 
PT.POL_VERSION_CD,
PT.EDWTRNS_TRANSEFFDT,
A.SUBM_ID,
A.PREMIUM_CHANGE_AMOUNT, 
A.PREMIUM_AMOUNT, 
A.COMMISS_AMT,
A.ONSET_IND,
A.SRC_SYS_CD,
A.CHARGE_CD,
T.TRANSACTION_GROUP_PART_NO,
T.TRANSACTION_GROUP_PART_CNT,
T.SOURCE_PROCESS_TS,
A.*

FROM ADW.DIVISIONAL_PREMIUMS A
INNER JOIN EDW.EDW_TRANS T on T.EDW_TRANS_ID=A.EDW_TRANS_ID
INNER JOIN qdm.VW_TRANS_POLTERM PT on A.EDW_TRANS_ID=PT.EDW_TRANS_ID 
WHERE PT.POL_NO_CD IN ('&POL_NO_CD')
AND PT.POL_MODULE_CD IN ('&POL_MODULE_CD')
AND PT.POL_VERSION_CD IN ('&POL_VERSION_CD')
AND A.SUBM_ID IN ('&SUBM_ID')
AND CHARGE_CD IS NULL
ORDER BY A.SUBM_ID DESC
