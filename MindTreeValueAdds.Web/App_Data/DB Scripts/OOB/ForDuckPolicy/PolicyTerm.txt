PT.* ,PT.CANCEL_NON_RENEW_CD,CANCEL_CATG_LONG_DESC, PT.DEC_PAGE_NM_TXT
FROM qdm.VW_TRANS_POLTERM PT
INNER JOIN EDW.EDW_TRANS T
ON T.EDW_TRANS_ID=PT.EDW_TRANS_ID
WHERE PT.POL_NO_CD IN ('&POL_NO_CD')
AND PT.POL_MODULE_CD IN ('&POL_MODULE_CD')
AND PT.POL_VERSION_CD IN ('&POL_VERSION_CD')
