SELECT q.POLICY_NO,
q.POLICY_MOD,
q.POLICY_VERSION,
subm.SOURCE_SUBMISSION_ID,
subm.TRANS_TYPE_CD,
q.ENDORSEMENT_NO,
subm.TRANS_SUBTYPE_CD,
q.TOT_APRPP_PREM_AMT,    --this is total APRP . Should be equal to APRP taxes+non taxes. Premium change in duck
q.TOT_APRP_PRM_EXCLD_TX_SRCHRG,    --this is APRP not including taxes. BDM report uses this.
q.TOT_APRP_PREM_TX_SRCHRG_AMT,    --this is APRP only for taxes. BDM report uses this
q.TOT_PRORATD_PREM_TX_SRCHRG_AMT,   --this is PRP shown from Duck Premium Written
subm.ANNUALIZED_EXPIRING_PREM_AMT, q.CURRENT_ANNUALIZED_PREM_AMT,     --form updating cube screen values
subm.ROW_CONDITION_CD,   --A,C, D- means Open, Lost, Deleted respectively
subm.*, q.*
FROM BDM_EDP.VW_Submission subm
INNER JOIN  BDM_EDP.VW_QUOTE q
ON subm.submission_ID=q.submission_ID
WHERE Q.TOT_POL_PREM_AMT IS NOT NULL
AND subm.load_DT>To_Date('02-FEB-20','dd-MON-yy')
AND q.POLICY_NO in ('&POL_NO_CD')
AND q.POLICY_MOD in ('&POL_MODULE_CD')
AND q.POLICY_VERSION in ('&POL_VERSION_CD')
AND subm.SOURCE_SUBMISSION_ID IN ('&SOURCE_SUBMISSION_ID')
AND rownum<5000
ORDER BY
q.POLICY_NO,
q.POLICY_MOD,
q.POLICY_VERSION,
subm.SOURCE_SUBMISSION_ID ASC
