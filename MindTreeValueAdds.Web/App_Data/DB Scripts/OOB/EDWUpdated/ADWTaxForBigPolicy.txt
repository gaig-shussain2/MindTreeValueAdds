select POL_NO_CD, POL_MODULE_CD, POL_VERSION_CD, SUBM_ID, sum(PREMIUM_CHANGE_AMOUNT) PREMIUM_CHANGE_AMOUNT from 
(
	select PT.BUSNACTYCD, PT.ENDT_NO, PT.SYSACTYCD, PT.POL_NO_CD, PT.POL_MODULE_CD, PT.POL_VERSION_CD,PT.EDWTRNS_TRANSEFFDT,
	A.SUBM_ID,
	A.PREMIUM_CHANGE_AMOUNT,   
	A.PREMIUM_AMOUNT,
	A.COMMISS_AMT,
	A.ONSET_IND,
	A.SRC_SYS_CD,
	A.CHARGE_CD,
	T.TRANSACTION_GROUP_PART_NO,
	T.TRANSACTION_GROUP_PART_CNT,
	T.SOURCE_PROCESS_TS
	from ADW.DIVISIONAL_PREMIUMS A 
	inner join EDW.EDW_TRANS T on T.EDW_TRANS_ID=A.EDW_TRANS_ID
	inner join qdm.VW_TRANS_POLTERM PT on A.EDW_TRANS_ID=PT.EDW_TRANS_ID 
	where PT.POL_NO_CD ='&POL_NO_CD' and PT.POL_MODULE_CD = '&POL_MODULE_CD' and PT.POL_VERSION_CD = '&POL_VERSION_CD' and CHARGE_CD is not null and REINS_CESSION_ID = '0' order by A.SUBM_ID desc
)
group by POL_NO_CD, POL_MODULE_CD, POL_VERSION_CD, SUBM_ID
